Gate G3 Pseudocode: Taker Flow Gate (NEW)

# G3: CoinGlass Taker Flow Gate (replaces old 3+ veto)
def taker_flow_gate(direction, cg_snapshot):
    # 1. Check freshness (hard kill if stale)
    if cg_snapshot.age_ms > 120_000:
        return SKIP("CG data stale")

    # 2. Calculate taker alignment
    taker_total = cg.taker_buy_volume + cg.taker_sell_volume
    if taker_total == 0:
        return PASS  # no data

    if direction == "UP":
        taker_aligned = (cg.taker_buy_volume / taker_total) > 0.55
        taker_opposing = (cg.taker_sell_volume / taker_total) > 0.55
    else:
        taker_aligned = (cg.taker_sell_volume / taker_total) > 0.55
        taker_opposing = (cg.taker_buy_volume / taker_total) > 0.55

    # 3. Check smart money alignment
    if direction == "UP":
        smart_opposing = cg.top_position_short_pct > 52
    else:
        smart_opposing = cg.top_position_long_pct > 52

    # 4. Decision matrix
    if taker_opposing and smart_opposing:
        return SKIP("both opposing — 58% WR bucket")  # HARD KILL

    if taker_opposing:
        return RAISE_THRESHOLD(+0.05)  # 73% WR — marginal, need stronger DUNE

    if taker_aligned:
        return LOWER_THRESHOLD(-0.02)  # 80%+ WR — CG confirmation bonus

    return PASS  # neutral — no modifier
